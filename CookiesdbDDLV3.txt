CREATE TABLE Customer
(
   	CustomerID INT NOT NULL IDENTITY(3000,1) PRIMARY KEY,
   	CustFName VARCHAR(15) NOT NULL,
   	CustLName VARCHAR (20) NOT NULL,
   	CustEmail VARCHAR (40),
	CustPhone VARCHAR(16),
);
CREATE TABLE Ingredients
(
   	IngID INT NOT NULL IDENTITY(100,1) PRIMARY KEY,
   	IngName VARCHAR (30) NOT NULL,  	
);

CREATE TABLE ProductTypes
(
   	PTypeID INT NOT NULL IDENTITY PRIMARY KEY,
	PTypeDesc VARCHAR(35),
);


CREATE TABLE Promotions
(
   	PromoID INT NOT NULL IDENTITY(200,1) PRIMARY KEY,
	PromoDesc VARCHAR(200),
	DiscountAmt DECIMAL(2,2),
);

CREATE TABLE Products
(
   	ProductID INT NOT NULL IDENTITY(2000,1) PRIMARY KEY,
   	PTypeID INT NOT NULL REFERENCES ProductTypes(PTypeID),
   	ProductName VARCHAR(15),
   	Price MONEY,
	WholesalePrice AS (Price * .80)
);

CREATE TABLE Recipe
(
   	IngID INT NOT NULL REFERENCES Ingredients(IngID),
   	ProductID INT NOT NULL REFERENCES Products(ProductID),
	Measure DECIMAL(5,2),
	Quantity INT,
   	PRIMARY KEY (IngID, ProductID)
);
CREATE TABLE Orders
(
   	OrderID INT NOT NULL IDENTITY(400,1) PRIMARY KEY,
   	CustomerID INT NOT NULL REFERENCES Customer(CustomerID),
   	OrderDate DATETIME,
);

CREATE TABLE OrderItems
(
   	OrderID INT NOT NULL REFERENCES Orders(OrderID),
   	ProductID INT NOT NULL REFERENCES Products(ProductID),
	PromoID INT NOT NULL REFERENCES Promotions(PromoID),
	Quantity INT,
	Price MONEY,
	PRIMARY KEY (OrderID, ProductID)
);

CREATE TABLE dwProductsDim
(
   	ProductID INT PRIMARY KEY,
	ProductName VARCHAR(15),
	UnitPrice MONEY,
	UnitWhlSlePrice MONEY,
	RecTimestamp DATETIME
);

CREATE TABLE dwCustomerDim
(
   	CustomerID INT PRIMARY KEY,
	CustomerFName VARCHAR(15),
	CustomerLName VARCHAR(20),
	CustEmail VARCHAR(40),
	CustPhone VARCHAR(16),
	RecTimestamp DATETIME

);

CREATE TABLE dwOrdersAgg
(
   	OrderID INT PRIMARY KEY,
	TotalSalesPrice MONEY,
	TotalWhlSlePrice MONEY,
	TotalItems INT,
	RecTimestamp DATETIME
);

CREATE TABLE dwPromotionDim
(
   	PromoID INT PRIMARY KEY,
	PromoDesc VARCHAR(200),
	DiscountAmt DECIMAL(2,2),
	RecTimestamp DATETIME
);

CREATE TABLE dwDateDim
(
   	DateID INT IDENTITY(1,1) PRIMARY KEY,
	RawDate DATE,
	HourOfDay INT,
	[DayOfWeek] VARCHAR(15),
	MonthOfYear VARCHAR(15),
	QtrOfYear VARCHAR(15),
	PurchYear INT,
	RecTimestamp DATETIME
);
